# PythonAnywhere Quick Commands Reference

## Initial Setup Commands (Run Once)

# 1. Clone repository
git clone https://github.com/bandaroyce88/LoanVista.git
cd LoanVista

# 2. Create virtual environment
mkvirtualenv --python=/usr/bin/python3.10 loanvista-env

# 3. Install dependencies
pip install -r requirements.txt

# 4. Create .env file
nano .env
# (Add your credentials, then Ctrl+X, Y, Enter to save)

# 5. Load environment variables
export $(cat .env | xargs)

# 6. Run migrations
python manage.py migrate

# 7. Create superuser
python manage.py createsuperuser

# 8. Collect static files
python manage.py collectstatic --noinput

# 9. Setup notifications
python setup_notification_templates.py

---

## Update Commands (When You Make Changes)

# Pull latest code
cd LoanVista
git pull origin main

# Activate environment
workon loanvista-env

# Install new dependencies (if any)
pip install -r requirements.txt

# Run migrations (if any)
python manage.py migrate

# Collect static files
python manage.py collectstatic --noinput

# Reload web app
touch /var/www/bandaroyce88_pythonanywhere_com_wsgi.py
